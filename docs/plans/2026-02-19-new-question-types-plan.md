### Task 3: Create multi-select UI component

**Files:**
- Create: `packages/web/src/components/question-types/multi-select.tsx`

**Step 1: Create the component**

Model after `multiple-choice.tsx` but use checkboxes and all-or-nothing scoring. The component receives a `question` with `options`, renders checkboxes, and on submit checks that the selected set matches the correct set exactly.

```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { cn } from "@/lib/utils";
import { RichContent } from "@/components/rich-content";

interface QuestionOption {
  id: number;
  questionId: number;
  optionText: string;
  isCorrect: boolean;
}

interface Question {
  id: number;
  type: string;
  question: string;
  options: QuestionOption[];
}

interface MultiSelectProps {
  question: Question;
  onAnswer: (isCorrect: boolean, userAnswer: string) => void;
  disabled: boolean;
}

export function MultiSelect({ question, onAnswer, disabled }: MultiSelectProps) {
  const [selected, setSelected] = useState<Set<number>>(new Set());

  const handleToggle = (optionId: number) => {
    if (disabled) return;
    setSelected(prev => {
      const next = new Set(prev);
      if (next.has(optionId)) next.delete(optionId);
      else next.add(optionId);
      return next;
    });
  };

  const handleSubmit = () => {
    if (selected.size === 0) return;

    const isCorrect = question.options.every(
      (o) => o.isCorrect === selected.has(o.id)
    );
    const selectedTexts = question.options
      .filter((o) => selected.has(o.id))
      .map((o) => o.optionText);
    onAnswer(isCorrect, JSON.stringify(selectedTexts));
  };

  return (
    <div className="space-y-4">
      <p className="text-sm text-muted-foreground">Select all that apply:</p>
      <div className="space-y-2">
        {question.options.map((option) => (
          <button
            key={option.id}
            onClick={() => handleToggle(option.id)}
            disabled={disabled}
            className={cn(
              "w-full rounded-lg border p-4 text-left transition-all",
              "hover:border-primary focus-visible:border-ring focus-visible:ring-ring/50 focus-visible:ring-[3px]",
              selected.has(option.id) && "border-primary bg-primary/10 font-medium",
              disabled && "cursor-not-allowed opacity-75",
              disabled && option.isCorrect && "border-green-500 bg-green-500/10",
              disabled && selected.has(option.id) && !option.isCorrect && "border-destructive bg-destructive/10"
            )}
          >
            <div className="flex items-center gap-3">
              <div
                className={cn(
                  "flex size-5 shrink-0 items-center justify-center rounded border-2",
                  selected.has(option.id)
                    ? "border-primary bg-primary"
                    : "border-muted-foreground/50"
                )}
              >
                {selected.has(option.id) && (
                  <svg className="size-3 text-primary-foreground" viewBox="0 0 12 12" fill="none">
                    <path d="M2 6l3 3 5-5" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" />
                  </svg>
                )}
              </div>
              <RichContent content={option.optionText} className="flex-1" />
            </div>
          </button>
        ))}
      </div>

      {!disabled && (
        <Button onClick={handleSubmit} disabled={selected.size === 0} className="w-full">
          Submit Answer
        </Button>
      )}
    </div>
  );
}
```

### Task 4: Create cloze deletion UI component

**Files:**
- Create: `packages/web/src/components/question-types/cloze.tsx`

**Step 1: Create the component**

The cloze component parses `{{c1::word}}` and `{{c1::word::hint}}` markers from `correctAnswer.text`. It shows the text with blanks for the active cloze group and text inputs in place of hidden words. Other cloze groups (not currently tested) show their words normally.

```tsx
"use client";

import { useState, useMemo } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { cn } from "@/lib/utils";
import { RichContent } from "@/components/rich-content";

interface Question {
  id: number;
  type: string;
  question: string;
  correctAnswer: string | null;
}

interface ClozeProps {
  question: Question;
  onAnswer: (isCorrect: boolean, userAnswer: string) => void;
  disabled: boolean;
}

interface ClozeGroup {
  id: string;
  blanks: { answer: string; hint?: string }[];
}

function parseCloze(text: string): { groups: ClozeGroup[]; groupIds: string[] } {
  const regex = /\{\{c(\d+)::([^}]*)\}\}/g;
  const groupMap = new Map<string, { answer: string; hint?: string }[]>();
  let match;
  while ((match = regex.exec(text)) !== null) {
    const groupId = `c${match[1]}`;
    const parts = match[2].split("::");
    const answer = parts[0];
    const hint = parts[1] || undefined;
    if (!groupMap.has(groupId)) groupMap.set(groupId, []);
    groupMap.get(groupId)!.push({ answer, hint });
  }
  const groupIds = [...groupMap.keys()].sort((a, b) => {
    const numA = parseInt(a.slice(1));
    const numB = parseInt(b.slice(1));
    return numA - numB;
  });
  const groups = groupIds.map(id => ({ id, blanks: groupMap.get(id)! }));
  return { groups, groupIds };
}

function renderClozeText(
  text: string,
  activeGroup: string,
  answers: string[],
  disabled: boolean,
  onAnswerChange: (index: number, value: string) => void
): React.ReactNode[] {
  const parts: React.ReactNode[] = [];
  const regex = /\{\{c(\d+)::([^}]*)\}\}/g;
  let lastIndex = 0;
  let blankIndex = 0;
  let match;

  while ((match = regex.exec(text)) !== null) {
    // Text before this match
    if (match.index > lastIndex) {
      parts.push(<span key={`text-${lastIndex}`}>{text.slice(lastIndex, match.index)}</span>);
    }

    const groupId = `c${match[1]}`;
    const answerParts = match[2].split("::");
    const answer = answerParts[0];
    const hint = answerParts[1] || undefined;

    if (groupId === activeGroup) {
      const idx = blankIndex++;
      if (disabled) {
        const userAnswer = answers[idx] || "";
        const isCorrect = userAnswer.trim().toLowerCase() === answer.trim().toLowerCase();
        parts.push(
          <span
            key={`blank-${idx}`}
            className={cn(
              "inline-block rounded px-2 py-0.5 font-medium mx-0.5",
              isCorrect ? "bg-green-500/20 text-green-700 dark:text-green-300" : "bg-destructive/20 text-destructive"
            )}
          >
            {userAnswer || "___"} {!isCorrect && <span className="text-xs opacity-70">({answer})</span>}
          </span>
        );
      } else {
        parts.push(
          <Input
            key={`input-${idx}`}
            className="inline-block w-32 mx-1 h-7 text-sm"
            placeholder={hint || "..."}
            value={answers[idx] || ""}
            onChange={(e) => onAnswerChange(idx, e.target.value)}
            autoFocus={idx === 0}
          />
        );
      }
    } else {
      // Show the word for non-active groups
      parts.push(<span key={`reveal-${match.index}`} className="font-medium">{answer}</span>);
    }

    lastIndex = match.index + match[0].length;
  }

  // Trailing text
  if (lastIndex < text.length) {
    parts.push(<span key={`text-end`}>{text.slice(lastIndex)}</span>);
  }

  return parts;
}

export function Cloze({ question, onAnswer, disabled }: ClozeProps) {
  const parsed = useMemo(() => {
    if (!question.correctAnswer) return null;
    try {
      const data = JSON.parse(question.correctAnswer) as { text: string };
      return parseCloze(data.text);
    } catch {
      return null;
    }
  }, [question.correctAnswer]);

  const [activeGroupIndex, setActiveGroupIndex] = useState(0);
  const [answers, setAnswers] = useState<string[]>([]);
  const [allResults, setAllResults] = useState<{ correct: number; total: number }>({ correct: 0, total: 0 });
  const [submitted, setSubmitted] = useState(false);

  if (!parsed || parsed.groups.length === 0) {
    return <div className="text-muted-foreground">Invalid cloze question.</div>;
  }

  const activeGroup = parsed.groups[activeGroupIndex];
  const isLastGroup = activeGroupIndex === parsed.groups.length - 1;

  const handleAnswerChange = (index: number, value: string) => {
    const next = [...answers];
    next[index] = value;
    setAnswers(next);
  };

  const handleSubmit = () => {
    // Check current group
    let groupCorrect = 0;
    activeGroup.blanks.forEach((blank, i) => {
      const userAnswer = (answers[i] || "").trim().toLowerCase();
      if (userAnswer === blank.answer.trim().toLowerCase()) groupCorrect++;
    });

    const newResults = {
      correct: allResults.correct + groupCorrect,
      total: allResults.total + activeGroup.blanks.length,
    };
    setAllResults(newResults);
    setSubmitted(true);

    if (isLastGroup) {
      // All groups done — report final result
      const isCorrect = newResults.correct === newResults.total;
      onAnswer(isCorrect, JSON.stringify({ answers, groupResults: newResults }));
    }
  };

  const handleNextGroup = () => {
    setActiveGroupIndex(activeGroupIndex + 1);
    setAnswers([]);
    setSubmitted(false);
  };

  const clozeText = question.correctAnswer ? JSON.parse(question.correctAnswer).text : "";

  return (
    <div className="space-y-4">
      {parsed.groups.length > 1 && (
        <p className="text-sm text-muted-foreground">
          Group {activeGroupIndex + 1} of {parsed.groups.length}
        </p>
      )}
      <div className="rounded-lg border p-4 text-base leading-relaxed flex flex-wrap items-center gap-y-2">
        {renderClozeText(clozeText, activeGroup.id, answers, disabled || submitted, handleAnswerChange)}
      </div>

      {!disabled && !submitted && (
        <Button
          onClick={handleSubmit}
          disabled={answers.length === 0 || answers.every(a => !a.trim())}
          className="w-full"
        >
          Check Answer{activeGroup.blanks.length > 1 ? "s" : ""}
        </Button>
      )}

      {submitted && !isLastGroup && !disabled && (
        <Button onClick={handleNextGroup} className="w-full">
          Next Group ({activeGroupIndex + 2} of {parsed.groups.length})
        </Button>
      )}
    </div>
  );
}
```

### Task 5: Create code evaluation UI component

**Files:**
- Create: `packages/web/src/components/question-types/code-eval.tsx`

**Step 1: Create the component**

The component parses `correctAnswer` JSON to get `code`, `language`, `mode`. In auto mode, it works like free_text. In AI mode, it works like open_ended (calls `/api/chat/evaluate`).

```tsx
"use client";

import { useState, useMemo } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { RichContent } from "@/components/rich-content";
import { Loader2 } from "lucide-react";

interface Question {
  id: number;
  type: string;
  question: string;
  correctAnswer: string | null;
  explanation: string | null;
}

interface CodeEvalProps {
  question: Question;
  onAnswer: (isCorrect: boolean, userAnswer: string) => void;
  disabled: boolean;
}

interface CodeEvalData {
  code: string;
  language: string;
  mode: "auto" | "ai";
  accepted?: string[];
  referenceAnswer?: string;
}

export function CodeEval({ question, onAnswer, disabled }: CodeEvalProps) {
  const data = useMemo<CodeEvalData | null>(() => {
    if (!question.correctAnswer) return null;
    try {
      return JSON.parse(question.correctAnswer) as CodeEvalData;
    } catch {
      return null;
    }
  }, [question.correctAnswer]);

  const [answer, setAnswer] = useState("");
  const [evaluating, setEvaluating] = useState(false);
  const [feedback, setFeedback] = useState<string | null>(null);

  if (!data) {
    return <div className="text-muted-foreground">Invalid code evaluation question.</div>;
  }

  const codeBlock = `\`\`\`${data.language}\n${data.code}\n\`\`\``;

  const handleAutoSubmit = () => {
    if (!answer.trim() || !data.accepted) return;
    const normalized = answer.trim().toLowerCase();
    const isCorrect = data.accepted.some(
      (a) => a.trim().toLowerCase() === normalized
    );
    onAnswer(isCorrect, answer.trim());
  };

  const handleAiSubmit = async () => {
    if (!answer.trim() || evaluating) return;
    setEvaluating(true);

    try {
      const res = await fetch("/api/chat/evaluate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          questionId: question.id,
          userAnswer: answer.trim(),
        }),
      });

      if (!res.ok) {
        throw new Error("Evaluation failed");
      }

      const result = await res.json();
      setFeedback(result.feedback);
      onAnswer(result.correct, answer.trim());
    } catch (error) {
      console.error("Evaluation error:", error);
      setFeedback("Could not evaluate your answer. Please review manually.");
      onAnswer(false, answer.trim());
    } finally {
      setEvaluating(false);
    }
  };

  const handleSubmit = data.mode === "auto" ? handleAutoSubmit : handleAiSubmit;

  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === "Enter" && !disabled && data.mode === "auto") {
      handleAutoSubmit();
    }
  };

  return (
    <div className="space-y-4">
      <div className="rounded-lg overflow-hidden">
        <RichContent content={codeBlock} />
      </div>

      {data.mode === "auto" ? (
        <Input
          type="text"
          placeholder="Type your answer..."
          value={answer}
          onChange={(e) => setAnswer(e.target.value)}
          onKeyDown={handleKeyDown}
          disabled={disabled}
          className="w-full"
          autoFocus
        />
      ) : (
        <textarea
          value={answer}
          onChange={(e) => setAnswer(e.target.value)}
          placeholder="Write your answer..."
          disabled={disabled || evaluating}
          rows={4}
          className="w-full resize-none rounded-md border bg-background px-3 py-2 text-sm placeholder:text-muted-foreground focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-ring disabled:opacity-50"
          autoFocus
        />
      )}

      {!disabled && !evaluating && (
        <Button onClick={handleSubmit} disabled={!answer.trim()} className="w-full">
          Submit Answer
        </Button>
      )}

      {evaluating && (
        <Button disabled className="w-full">
          <Loader2 className="mr-2 h-4 w-4 animate-spin" />
          AI is evaluating your answer...
        </Button>
      )}

      {feedback && (
        <div className="rounded-lg border bg-muted/50 p-3">
          <p className="text-sm font-medium mb-1">AI Feedback:</p>
          <p className="text-sm text-muted-foreground">{feedback}</p>
        </div>
      )}

      {disabled && data.mode === "auto" && (
        <div className="rounded-lg border bg-muted/50 p-3">
          <p className="text-sm">
            <span className="font-medium">Your answer:</span> {answer}
          </p>
        </div>
      )}
    </div>
  );
}
```
